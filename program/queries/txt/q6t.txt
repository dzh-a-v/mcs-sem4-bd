SELECT
    g.galaxy_id,
    g.conf_plan,
    g.hab_plan,
    g.dist,
    'ground'::tele_type_enum AS tele_type,
    0 AS observed_stars_count
FROM galaxy g
LEFT JOIN (
    SELECT DISTINCT c.galaxy_id
    FROM cluster c
    JOIN stellar_system ss ON ss.cluster_id = c.cluster_id
    JOIN plan_system ps ON ps.stell_sys_id = ss.stellar_id
    JOIN star s ON s.sys_id = ps.system_id
    JOIN exoplanet e ON e.sys_id = ps.system_id
    JOIN observation o ON o.exo_id = e.exo_id
    JOIN telescope t ON t.tele_id = o.tele_id AND t.tele_type = 'ground'
) AS observed_galaxies ON observed_galaxies.galaxy_id = g.galaxy_id
WHERE observed_galaxies.galaxy_id IS NULL
ORDER BY g.galaxy_id;
